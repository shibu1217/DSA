//kth element 
#include<bits/stdc++.h>
using namespace std;

void solve(int arr[],int k,int n)
{
    priority_queue<int>maxh;
    
        for(int i=0;i<n;i++){
        maxh.push(arr[i]);
        if(maxh.size()>k)
        {
            maxh.pop();
        }
    }
    cout<< maxh.top();
    
}

int main()
{
    int arr[]={7,10,4,3,20,15};
    int k=3;
    int n=sizeof(arr)/sizeof(arr[0]);
    solve(arr,k,n);
    return 0;
}
